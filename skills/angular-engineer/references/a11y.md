# Accessibility (a11y) Patterns

## Quick Checklist (PR Review)

- [ ] Every interactive element reachable by keyboard alone
- [ ] Icon-only buttons/links have `aria-label` or `aria-labelledby`
- [ ] Form inputs associated to `<label>` (via `for`/`id` or `aria-labelledby`)
- [ ] Validation errors announced via `aria-live` or `LiveAnnouncer`
- [ ] Required fields have `aria-required="true"` and a visible indicator
- [ ] Dialogs trap focus on open; restore focus to the trigger on close
- [ ] Route changes announce the new page title via `LiveAnnouncer`
- [ ] Color is never the sole differentiator (contrast ≥ 4.5:1 normal text)
- [ ] `@defer` blocks use `aria-busy` or a live region so AT sees loaded content
- [ ] `axe-core` runs in CI; fails on any `critical` or `serious` violation

### FocusMonitor — keyboard vs mouse vs touch origin

```typescript
// src/app/shared/components/custom-button/custom-button.component.ts
import { FocusMonitor, FocusOrigin } from '@angular/cdk/a11y';
import { ElementRef, OnDestroy, inject, signal } from '@angular/core';

export class CustomButtonComponent implements OnDestroy {
  private fm    = inject(FocusMonitor);
  private elRef = inject(ElementRef);
  focusOrigin   = signal<FocusOrigin>(null); // null = blurred; 'keyboard' | 'mouse' | 'touch' when focused

  constructor() { this.fm.monitor(this.elRef, true).subscribe(o => this.focusOrigin.set(o)); }
  ngOnDestroy(): void { this.fm.stopMonitoring(this.elRef); }
}
```

## 2. ARIA Patterns for Angular Material Components

### Icon-only button

```html
<!-- ❌ BAD — no accessible name -->
<button mat-icon-button (click)="delete(row)"><mat-icon>delete</mat-icon></button>

<!-- ✅ GOOD — dynamic label, icon hidden from AT -->
<button mat-icon-button [attr.aria-label]="'Delete ' + row.name" (click)="delete(row)">
  <mat-icon aria-hidden="true">delete</mat-icon>
</button>
```

### Dialog — title, description, initial focus

```typescript
// src/app/shared/dialogs/confirm-dialog/confirm-dialog.component.ts
@Component({
  template: `
    <h2 mat-dialog-title id="confirm-title">{{ data.title }}</h2>
    <mat-dialog-content><p id="confirm-desc">{{ data.message }}</p></mat-dialog-content>
    <mat-dialog-actions align="end">
      <button mat-button mat-dialog-close>Cancel</button>
      <button mat-button [mat-dialog-close]="true" cdkFocusInitial>Confirm</button>
    </mat-dialog-actions>
  `,
})
export class ConfirmDialogComponent {
  data = inject(MAT_DIALOG_DATA);
}

// Wire ariaLabelledBy / ariaDescribedBy when opening:
this.dialog.open(ConfirmDialogComponent, {
  ariaLabelledBy: 'confirm-title',
  ariaDescribedBy: 'confirm-desc',
  data: { title: 'Delete item?', message: 'This action cannot be undone.' },
});
```

### Snackbar — politeness level

```typescript
// src/app/core/services/notification.service.ts
@Injectable({ providedIn: 'root' })
export class NotificationService {
  private snackBar = inject(MatSnackBar);

  success(msg: string): void {
    this.snackBar.open(msg, 'Close', { duration: 4000, politeness: 'polite' });
  }
  error(msg: string): void {
    // assertive interrupts the current SR sentence — reserve for errors
    this.snackBar.open(msg, 'Close', { duration: 0, politeness: 'assertive' });
  }
}
```

## 3. Keyboard Navigation

### Focus management after dialog close

```typescript
// src/app/features/users/user-list/user-list.component.ts
@Component({ template: `<button #addBtn mat-raised-button (click)="open()">Add User</button>` })
export class UserListComponent {
  @ViewChild('addBtn', { read: ElementRef }) addBtn!: ElementRef<HTMLButtonElement>;
  private dialog = inject(MatDialog);

  open(): void {
    this.dialog.open(AddUserDialogComponent)
      .afterClosed()
      .subscribe(() => this.addBtn.nativeElement.focus());
  }
}
```

### Custom overlay keyboard trap (`cdkTrapFocus`)

```typescript
// src/app/shared/components/side-panel/side-panel.component.ts
import { A11yModule } from '@angular/cdk/a11y';

@Component({
  standalone: true,

}
```

## 4. Screen Reader Announcements

### LiveAnnouncer for dynamic content

```typescript
// src/app/features/search/search-results/search-results.component.ts
import { LiveAnnouncer } from '@angular/cdk/a11y';

export class SearchResultsComponent {
  private announcer = inject(LiveAnnouncer);
  onLoaded(count: number): void { this.announcer.announce(`${count} results found`, 'polite'); }
  onError(): void { this.announcer.announce('Search failed. Please try again.', 'assertive'); }
}
```

### Route change announcements

```typescript
// src/app/core/services/route-announcer.service.ts
import { LiveAnnouncer } from '@angular/cdk/a11y';
import { Router, NavigationEnd } from '@angular/router';
import { Title } from '@angular/platform-browser';

@Injectable({ providedIn: 'root' })
export class RouteAnnouncerService {
  private router = inject(Router); private announcer = inject(LiveAnnouncer); private title = inject(Title);

  init(): void {
    this.router.events.pipe(filter(e => e instanceof NavigationEnd))
      .subscribe(() => this.announcer.announce(`Navigated to ${this.title.getTitle()}`, 'polite'));
  }
}
// app.component.ts — call once at startup
export class AppComponent { constructor(ra: RouteAnnouncerService) { ra.init(); } }
```

## 5. Form Accessibility

```typescript
// src/app/features/auth/login/login.component.ts
import { LiveAnnouncer } from '@angular/cdk/a11y';

export class LoginFormComponent {
  private announcer = inject(LiveAnnouncer);
  onSubmit(): void {
    if (this.form.invalid) {
      this.form.markAllAsTouched();
      const n = Object.values(this.form.controls).filter(c => c.invalid).length;
      this.announcer.announce(`Form has ${n} error${n > 1 ? 's' : ''}. Please review.`, 'assertive');
    }
  }
}
```

```html
<!-- src/app/features/auth/login/login.component.html -->
<!-- mat-form-field wires aria-labelledby automatically -->
<mat-form-field>
  <mat-label>Email address</mat-label>
  <input matInput type="email" formControlName="email" autocomplete="email"
         aria-required="true"
         [attr.aria-describedby]="emailCtrl.invalid && emailCtrl.touched ? 'email-err' : null" />
  <mat-error id="email-err" aria-live="polite">
    <ng-container *ngIf="emailCtrl.hasError('required')">Email is required.</ng-container>
    <ng-container *ngIf="emailCtrl.hasError('email')">Enter a valid email.</ng-container>
  </mat-error>
</mat-form-field>

<!-- Required indicator: hide asterisk from AT; explain in visually-hidden text -->
<mat-form-field>
  <mat-label>Full name <span aria-hidden="true">*</span></mat-label>
  <input matInput formControlName="fullName" aria-required="true" />
</mat-form-field>
<p class="visually-hidden">Fields marked with * are required.</p>
```

```scss
// src/styles/_a11y.scss
.visually-hidden {
  position: absolute; width: 1px; height: 1px;
  padding: 0; margin: -1px; overflow: hidden;
  clip: rect(0, 0, 0, 0); white-space: nowrap; border: 0;
}
```

## 6. Angular CDK ListKeyManager

```typescript
// src/app/shared/components/custom-menu/custom-menu.component.ts
import { ActiveDescendantKeyManager, Highlightable } from '@angular/cdk/a11y';
import { AfterViewInit, Component, HostListener, Input, QueryList, ViewChildren } from '@angular/core';

// Each option implements Highlightable so the key manager can mark it active
@Component({ selector: 'app-menu-item', template: `<li role="option" [class.active]="isActive" [id]="id"><ng-content /></li>` })
export class MenuItemComponent implements Highlightable {
  @Input() id!: string; isActive = false;
  setActiveStyles(): void   { this.isActive = true; }
  setInactiveStyles(): void { this.isActive = false; }
}

@Component({
  selector: 'app-custom-menu',
  template: `
    <ul role="listbox" tabindex="0"
        [attr.aria-activedescendant]="keyManager.activeItem?.id ?? null"
        (focus)="keyManager.setFirstItemActive()">
      <app-menu-item *ngFor="let item of items; let i = index" [id]="'opt-' + i">{{ item.label }}</app-menu-item>
    </ul>
  `,
})
export class CustomMenuComponent implements AfterViewInit {
  @Input() items: { label: string }[] = [];
  @ViewChildren(MenuItemComponent) menuItems!: QueryList<MenuItemComponent>;
  keyManager!: ActiveDescendantKeyManager<MenuItemComponent>;

  ngAfterViewInit(): void {
    this.keyManager = new ActiveDescendantKeyManager(this.menuItems).withWrap().withTypeAhead();
  }
  @HostListener('keydown', ['$event'])
  onKeydown(e: KeyboardEvent): void { this.keyManager.onKeydown(e); }
}
```

## 7. `@defer` and a11y

```typescript
// src/app/features/dashboard/stats/stats.component.ts
import { LiveAnnouncer } from '@angular/cdk/a11y';

export class StatsComponent {
  private announcer = inject(LiveAnnouncer);
  statsLoaded = false;

  onLoaded(): void {
    this.statsLoaded = true;
    this.announcer.announce('Dashboard statistics loaded', 'polite');
  }
}
```

```html
<!-- src/app/features/dashboard/stats/stats.component.html -->
<!-- Wrap defer in aria-live so AT picks up inserted content -->
<section aria-live="polite" aria-label="Dashboard statistics" [attr.aria-busy]="!statsLoaded">
  @defer (on viewport) {
    <app-stats-chart (loaded)="onLoaded()" />
  } @placeholder {
    <div class="skeleton" aria-hidden="true"></div>   <!-- hidden — will be replaced -->
  } @error {
    <p role="alert">Failed to load statistics.</p>    <!-- role=alert = assertive live region -->
  }
</section>
```

`@loading` placeholders: `aria-hidden="true"` — they are transient. Announce completion via `LiveAnnouncer`, not DOM insertion alone.

## 8. Testing a11y

```bash
npm install --save-dev axe-core
```

```typescript
// src/app/features/auth/login/login.component.spec.ts  (Karma / Jasmine)
import { ComponentFixture, TestBed } from '@angular/core/testing';
import axe, { AxeResults } from 'axe-core';

describe('LoginFormComponent a11y', () => {
  let fixture: ComponentFixture<LoginFormComponent>;

  beforeEach(async () => {
    await TestBed.configureTestingModule({ imports: [LoginFormComponent] }).compileComponents();
    fixture = TestBed.createComponent(LoginFormComponent);
    fixture.detectChanges();
  });

  it('has no critical or serious axe violations', async () => {
    const { violations }: AxeResults = await axe.run(fixture.nativeElement, {
      runOnly: { type: 'tag', values: ['wcag2a', 'wcag2aa'] },
    });
    expect(violations.filter(v => ['critical', 'serious'].includes(v.impact as string))).toEqual([]);
  });
});
```

### CDK a11y testing utilities

```typescript
// src/app/shared/components/side-panel/side-panel.component.spec.ts
import { A11yModule } from '@angular/cdk/a11y';

describe('SidePanelComponent', () => {
  let fixture: ComponentFixture<SidePanelComponent>;
  beforeEach(async () => {
    await TestBed.configureTestingModule({ imports: [SidePanelComponent, A11yModule] }).compileComponents();
    fixture = TestBed.createComponent(SidePanelComponent);
    fixture.componentInstance.isOpen = true;
    fixture.detectChanges();
  });

  it('traps focus inside the panel', () => {
    const panel = fixture.nativeElement.querySelector('[cdkTrapFocus]');
    expect(panel.contains(document.activeElement)).toBeTrue();
  });
  it('has role=dialog and aria-modal', () => {
    const el = fixture.nativeElement.querySelector('[role="dialog"]');
    expect(el?.getAttribute('aria-modal')).toBe('true');
  });
});
```

### What to assert

| Concern | Assert |
|---|---|
| Interactive name | `aria-label` non-empty, or visible `<label>` associated |
| Error announcement | `aria-live` region text updates after `markAllAsTouched()` |
| Focus trap | `panel.contains(document.activeElement)` true after open |
| Focus restore | `document.activeElement === triggerElement` after close |
| Route announcement | `LiveAnnouncer.announce` spy called with new page title |
| No axe violations | `violations.filter(critical/serious)` is empty |
